package "calc"

expr :  _ sum ;

sum  : prod { PLUS prod | MINUS } ;

prod : elem { TIMES elem | DIVIDE elem } ;

elem : OPEN sum CLOSE | num ;

num  :  < number > _ ;

PLUS   : '+' _ ;
MINUS  : '-' _ ;
TIMES  : '*' _ ; 
DIVIDE : '/' _ ;
OPEN   : '(' _ ; 
CLOSE  : ')' _ ;

_      : { ' ' 
       |   '\t' } ;
